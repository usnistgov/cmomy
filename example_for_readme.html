Example usage

Example usage#

In [1]: import numpy as np

In [2]: import cmomy

In [3]: x = np.random.rand(100)

In [4]: m = x.mean()

In [5]: mom = np.array([((x - m) ** i).mean() for i in range(4)])

In [6]: c = cmomy.CentralMoments.from_vals(x, mom=3)

In [7]: mom
Out[7]: array([ 1.0000e+00, -2.1094e-17,  8.6709e-02,  1.2700e-03])

In [8]: c.cmom()
Out[8]: array([1.    , 0.    , 0.0867, 0.0013])

# break up into chunks
In [9]: c = cmomy.CentralMoments.from_vals(x.reshape(-1, 2), mom=3)

In [10]: c
Out[10]: 
<CentralMoments(val_shape=(2,), mom=(3,))>
array([[5.0000e+01, 4.9671e-01, 8.6679e-02, 1.9960e-03],
       [5.0000e+01, 4.8534e-01, 8.6675e-02, 5.4388e-04]])

In [11]: c.reduce(axis=0).cmom()
Out[11]: array([1.    , 0.    , 0.0867, 0.0013])

# unequal chunks
In [12]: x0, x1, x2 = x[:20], x[20:60], x[60:]

In [13]: cs = [cmomy.CentralMoments.from_vals(_, mom=3) for _ in (x0, x1, x2)]

In [14]: c = cs[0] + cs[1] + cs[2]

In [15]: c.cmom()
Out[15]: array([1.    , 0.    , 0.0867, 0.0013])
